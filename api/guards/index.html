Guards

Various TypeScript guards.

isArray​

function isArray<T>(value: T | ReadonlyArray<unknown>): value is GuardArray<T>
G.isArray([1, 2, 3]) // → true
pipe(['hello', 'world'], G.isArray) // → true

isBoolean​

function isBoolean<T>(value: T | boolean): value is GuardValue<T, boolean>
G.isBoolean(false) // → true
pipe(true, G.isBoolean) // → true

isDate​

function isDate<T>(value: T): value is Extract<T, Date>
G.isDate(new Date()) // → true
pipe(new Date(), G.isDate) // → true

isError​

function isError<T>(value: T | Error): value is GuardValue<T, Error>
G.isError(new Error('oops!')) // → true
pipe(new Error('oops!'), G.isError) // → true

isFunction​

function isFunction<T>(value: T | Function): value is GuardValue<T, Function>
G.isFunction(F.ignore) // → true
pipe(F.ignore, G.isFunction) // → true

isNot​

function isNot<T>(value: T, predicateFn: (value: T) => boolean): boolean
function isNot<T>(predicateFn: (value: T) => boolean): (value: T) => boolean
// ⬇️ const isNotString = <T>(value: T): value is Exclude<T, string> => G.isNot(value, G.isString)
isNotString(0) // → true

pipe('ts-belt', G.isNot(G.isString)) // → false

isNotNullable​

function isNotNullable<T>(value: T): value is NonNullable<T>
G.isNotNullable('ts-belt') // → true
G.isNotNullable(null) // → false
pipe(0, G.isNotNullable) // → true
pipe(undefined, G.isNotNullable) // → false

isNull​

function isNull<T>(value: T | null | undefined): value is null
G.isNull(null) // → true
pipe(null, G.isNull) // → true

isNullable​

function isNullable<T>(value: T): value is Extract<T, null | undefined>
G.isNullable(null) // → true
pipe(undefined, G.isNullable) // → true

isNumber​

function isNumber<T>(value: T | number): value is GuardValue<T, number>
G.isNumber(3) // → true
pipe(3, G.isNumber) // → true

isObject​

function isObject<T>(value: T | object): value is GuardObject<T>
G.isObject({}) // → true
pipe({}, G.isObject) // → true

isPromise​

function isPromise<T>(value: T | Promise<unknown>): value is GuardPromise<T>
G.isPromise(Promise.resolve(1)) // → true
pipe(Promise.resolve(1), G.isPromise) // → true

isString​

function isString<T>(value: T | string): value is GuardValue<T, string>
G.isString('ts') // → true
pipe('belt', G.isString) // → true

isUndefined​

function isUndefined<T>(value: T | null | undefined): value is undefined
G.isUndefined(undefined) // → true
pipe(undefined, G.isUndefined) // → true
